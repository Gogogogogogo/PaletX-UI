复选框-Checkbox

可以同时选择多个

代码示例 

    
      <div class="px-checkbox-group">
        <label class="px-checkbox-wrapper">
          <input type="checkbox" class="px-checkbox">
          <span>选项一</span>
        </label>
        <label class="px-checkbox-wrapper">
          <input type="checkbox" class="px-checkbox" checked>
          <span>选项二</span>
        </label>
        <label class="px-checkbox-wrapper">
          <input type="checkbox" class="px-checkbox" disabled>
          <span>选项三</span>
        </label>
        <label class="px-checkbox-wrapper">
          <input type="checkbox" class="px-checkbox" checked disabled>
          <span>选项四</span>
        </label>
      </div>
