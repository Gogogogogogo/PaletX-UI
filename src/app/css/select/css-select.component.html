下拉选择框-Select

提供两种不同的场景

场景一：不带输入模式

代码示例 

    
      <div ngbDropdown class="px-select">
        <input ngbDropdownToggle [(ngModel)]="inputSelect1" readonly placeholder="请选择">
        <i class="fa fa-caret-down"></i>
        <div class="dropdown-menu">
          <button (click)="inputSelect1 = '选项1'">选项1</button>
          <button (click)="inputSelect1 = '选项2'">选项2</button>
          <button (click)="inputSelect1 = '选项3'">选项3</button>
        </div>
      </div>
      <div ngbDropdown class="px-select">
        <input ngbDropdownToggle [(ngModel)]="inputSelect2" readonly placeholder="请选择" disabled>
        <i class="fa fa-caret-down"></i>
        <div class="dropdown-menu">
          <button (click)="inputSelect2 = '选项1'">选项1</button>
          <button (click)="inputSelect2 = '选项2'">选项2</button>
          <button (click)="inputSelect2 = '选项3'">选项3</button>
        </div>
      </div>

场景二：带输入模式

代码示例 

    
      <div ngbDropdown class="px-select addon">
        <input [(ngModel)]="inputSelect3" placeholder="请选择或输入">
        <a class="addon" role="button" ngbDropdownToggle></a>
        <div class="dropdown-menu">
          <button (click)="inputSelect3 = '选项1'">选项1</button>
          <button (click)="inputSelect3 = '选项2'">选项2</button>
          <button (click)="inputSelect3 = '选项3'">选项3</button>
        </div>
      </div>
